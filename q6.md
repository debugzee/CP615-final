# Mappings

Treat tasks as a multivalued attribute, so it becomes a separate table.
Also, the 1:N and 1:1 relationships are enforced using foreign keys and UNIQUE constraints.

```sql
CREATE TABLE Office (
    oID        VARCHAR(10) PRIMARY KEY,
    capacity   INT NOT NULL,
    CONSTRAINT chk_office_capacity CHECK (capacity > 0)
);

CREATE TABLE Department (
    deptID        VARCHAR(10) PRIMARY KEY,
    name          VARCHAR(100) NOT NULL,
    location      VARCHAR(100) NOT NULL,
    mainOfficeID  VARCHAR(10) NOT NULL UNIQUE,
    CONSTRAINT fk_department_office
        FOREIGN KEY (mainOfficeID) REFERENCES Office(oID)
);

CREATE TABLE Employee (
    empID      VARCHAR(10) PRIMARY KEY,
    fullName   VARCHAR(100) NOT NULL,
    salary     DECIMAL(10,2) NOT NULL,
    deptID     VARCHAR(10) NOT NULL,
    oID        VARCHAR(10) UNIQUE,
    CONSTRAINT chk_employee_salary CHECK (salary >= 0),
    CONSTRAINT fk_employee_department
        FOREIGN KEY (deptID) REFERENCES Department(deptID),
    CONSTRAINT fk_employee_office
        FOREIGN KEY (oID) REFERENCES Office(oID)
);

CREATE TABLE Task (
    empID   VARCHAR(10) NOT NULL,
    task    VARCHAR(200) NOT NULL,
    PRIMARY KEY (empID, task),
    CONSTRAINT fk_task_employee
        FOREIGN KEY (empID) REFERENCES Employee(empID)
        ON DELETE CASCADE
);
```
